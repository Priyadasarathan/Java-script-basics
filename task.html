function handleButtonClick(event) { // Get the button that was clicked const button = event.target; // Get the current value of the counter const counter = button.parentNode.querySelector('span'); let value = parseInt(counter.textContent); // Increment or decrement the counter based on the button that was clicked if (button.id === 'incrementBtn') { value++; } else if (button.id === 'decrementBtn') { value--; } // Update the counter display counter.textContent = value; }
